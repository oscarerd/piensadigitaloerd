Abrir consola powershell como administrador

usuario de conexión: curso1
comando para conectar con servidor:
	ssh -p 22 curso1@'dirección ip de un equipo en la red' 
Ejemplos:
	ssh -p 22 curso1@192.168.0.35
	ssh -p 22 curso1@192.168.0.33
